-- For문 컴포넌트 생성
function App() {
const weekArr = ["MON", "TUE", "WED", "THU", "FRI", "SAT", "SUN"];

const rendering = () => {
const result = [];
for (let i = 0; i < weekArr.length; i++) {
result.push(<span key={i}>{weekArr[i] + " / "}</span>);
}
return result;
};

return <div>{rendering()}</div>;
}
